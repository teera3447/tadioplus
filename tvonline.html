const M3U_SOURCE_URL = "https://raw.githubusercontent.com/teera3447/tadioplus/refs/heads/main/tadio1.m3u"; addEventListener('fetch', event => { event.respondWith(handleRequest(event.request)) }) async function handleRequest(request) { const url = new URL(request.url); // 1. ดึงรายการช่องสำหรับหน้าเว็บ if (url.pathname === "/list") { const res = await fetch(M3U_SOURCE_URL); const data = await res.text(); return new Response(data, { headers: { "Content-Type": "text/plain; charset=utf-8", "Access-Control-Allow-Origin": "*" } }); } // 2. ระบบ Proxy สำหรับ HTTP (ทำงานเมื่อหน้าเว็บส่ง ?url= มาให้) const targetUrl = url.searchParams.get('url'); if (targetUrl) { try { const decodedUrl = decodeURIComponent(targetUrl); const response = await fetch(decodedUrl, { headers: { "User-Agent": request.headers.get("User-Agent") || "Mozilla/5.0" }, redirect: 'follow' }); // สร้าง Response ใหม่โดยใช้ข้อมูลเดิมทั้งหมด (เพื่อรักษา Format วิดีโอและคุณภาพ) const proxyRes = new Response(response.body, response); // เพิ่ม CORS เพื่อให้ Browser ยอมรับการเล่นบน HTTPS proxyRes.headers.set("Access-Control-Allow-Origin", "*"); proxyRes.headers.delete("Content-Security-Policy"); proxyRes.headers.delete("X-Frame-Options"); return proxyRes; } catch (e) { return new Response("Error: " + e.message, { status: 500 }); } } return new Response("Tadio+ Proxy Active", { status: 200 }); }
